###### 示例代码 1

```vue
<template>
  <div style="height: 100%;width: 100%;">
    <div
      v-if="asNumber.length"
      style="height: 100%;width: 100%;"
      id="viz"
      ref="viz"
    ></div>
    <el-empty v-else :image-size="200" />
  </div>
</template>
<script setup name="Info">

import NeoVis from "neovis.js";

const props = defineProps({
  /* 数据 */
  asNumber: {
    type: String,
    default: "",
  },
});
const viz = ref(null);
onMounted(() => {
  console.log(props.asNumber, "asNumber");
  draw();
});
const draw = () => {
  let viz1 = null;
  var config = {
    containerId: "viz",
    neo4j: {
      // 社区版Neo4j只能使用DBMS的默认数据库，即neo4j
      serverUrl: "bolt://192.168.188.72:7687",
      serverUser: "neo4j",
      serverPassword: "sjwjabc",
    },
    visConfig: {
      // nodes: {
      //     shape: 'square'
      // },
      edges: {
        arrows: {
          to: { enabled: true },
        },
      },
    },
    // 节点
    labels: {
      relatives: {
        // label是需要显示在节点上的属性
        label: "name",
        value: "pagerank",
        // group: "community",
        // size是节点的大小，根据属性的值来决定
        size: "length",
        font: {
          color: "#606266",
          size: 25,
        },
      },
      asinfo: {
        label: "name",
        font: {
          size: 2,
          color: "F",
        },
        size: "26",
      },
    },
    // 关系
    relationships: {
      ipv4: {
        label: "name",
        thickness: "weight",
      },
      // LOCATED: {
      // 	label: "relation"
      // },
    },
    // arrows: true, //箭头

    // 初始的Cypher语句
    // MATCH (a:asinfo {name:'AS174'})-[]-(b) RETURN a,b
    // initialCypher: "MATCH (n:asinfo {name:'AS174'}) RETURN n LIMIT 25" //查询节点表的语句（查询asinfo表 节点名的name=AS174）
    // initialCypher: "MATCH (a)-[r:ipv4]->(b) WHERE a.name = 'AS701' RETURN a,r,b"  //查询关系表的语句（查询AS174节点的ipv4customer关系）
    initialCypher: `MATCH (a:asinfo {name:'${props.asNumber}'})-[r:ipv4]-(b) RETURN a,r,b`, //查询节点的语句（[这里边写的是关系名字]）
  };

  viz1 = new NeoVis(config);
  viz1._container = viz.value;
  viz1.render();

  // 点击节点出发的函数
  viz1.registerOnEvent("clickNode", (node) => {
    viz1.stabilize();

    //点击节点，使用updateWithCypher方法后，点击事件会多次挂载；
    // 所以每次执updateWithCypher方法之前要解绑之前绑定的点击事件this.viz._network.off('click')
    viz1.network.off("click");
    console.log("Node clicked: ", node.node["label"]);
    // props.asNumber.value = node.label
    const statement = `MATCH (a:asinfo {name:'${node.node["label"]}'})-[r:ipv4]-(b) RETURN a,r,b`;
    console.log(statement);
    // 点击节点，使用updateWithCypher方法后，点击事件会多次挂载；所以每次执updateWithCypher方法之前要解绑之前绑定的点击事件this.viz._network.off('click')
    // 使用Cypher语句的结果呈现新的可视化效果。Cypher查询中返回的任何Node和Relationship对象都将在可视化中呈现。当前不支持路径。
    viz1.renderWithCypher(statement);
  });
  console.log(viz1);
};

defineExpose({ draw });
</script>
```
